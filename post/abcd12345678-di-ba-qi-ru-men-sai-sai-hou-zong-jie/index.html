// 大整数加法
string add(string a, string b) {
    if (a.length() < b.length()) swap(a, b);
    
    string result = "";
    int carry = 0;
    int i = a.length() - 1;
    int j = b.length() - 1;
    
    while (i >= 0 || carry) {
        int sum = carry;
        if (i >= 0) sum += a[i] - '0';
        if (j >= 0) sum += b[j] - '0';
        
        result += char(sum % 10 + '0');
        carry = sum / 10;
        
        i--;
        j--;
    }
    
    reverse(result.begin(), result.end());
    return result;
}

// 大整数减法（a >= b）
string subtract(string a, string b) {
    string result = "";
    int borrow = 0;
    int i = a.length() - 1;
    int j = b.length() - 1;
    
    while (i >= 0) {
        int digit_a = a[i] - '0' - borrow;
        int digit_b = (j >= 0) ? b[j] - '0' : 0;
        
        if (digit_a < digit_b) {
            digit_a += 10;
            borrow = 1;
        } else {
            borrow = 0;
        }
        
        result += char(digit_a - digit_b + '0');
        
        i--;
        j--;
    }
    
    // 去除前导0
    while (result.length() > 1 && result.back() == '0') {
        result.pop_back();
    }
    
    reverse(result.begin(), result.end());
    return result;
}

// 大整数乘法
string multiply(string a, string b) {
    if (a == "0" || b == "0") return "0";
    
    int n = a.length();
    int m = b.length();
    vector<int> pos(n + m, 0);
    
    for (int i = n - 1; i >= 0; i--) {
        for (int j = m - 1; j >= 0; j--) {
            int mul = (a[i] - '0') * (b[j] - '0');
            int p1 = i + j;
            int p2 = i + j + 1;
            int sum = mul + pos[p2];
            
            pos[p2] = sum % 10;
            pos[p1] += sum / 10;
        }
    }
    
    string result;
    for (int num : pos) {
        if (!(result.empty() && num == 0)) {
            result += char(num + '0');
        }
    }
    
    return result.empty() ? "0" : result;
}

int main() {
    int k;
    string line;
    
    // 读取k
    if (!(cin >> k)) return 0;
    
    // 读取剩余的空格和换行符
    getline(cin, line); // 读取第一行剩下的部分
    
    // 读取第二行
    getline(cin, line);
    
    // 解析输入
    stringstream ss(line);
    string a, op, b;
    ss >> a >> op >> b;
    
    string result;
    
    if (op == "+") {
        result = add(a, b);
    } else if (op == "-") {
        result = subtract(a, b);
    } else if (op == "*") {
        result = multiply(a, b);
    }
    
    int len = result.length();
    
    // 输出前k位
    int front_len = min(k, len);
    cout << result.substr(0, front_len) << endl;
    
    // 输出后k位
    int back_len = min(k, len);
    int start_pos = len - back_len;
    if (start_pos < 0) start_pos = 0;
    
    string back_str = result.substr(start_pos, back_len);
    
    // 补前导0
    if ((int)back_str.length() < k) {
        back_str = string(k - back_str.length(), '0') + back_str;
    }
    
    cout << back_str << endl;
    
    return 0;
}

#include <iostream>
#include <cmath>
#include <algorithm>

using namespace std;

int main() {
    double a;
    cin >> a; // 读取输入（最多一位小数，0<a<100）
    
    // 0<a<100范围内的所有完全平方数（1²~10²）
    const int squares[] = {1, 4, 9, 16, 25, 36, 49, 64, 81, 100};
    const int square_count = sizeof(squares) / sizeof(squares[0]);
    
    double min_diff = 100.0; // 初始化为超过范围的极大值
    for (int sq : squares) {
        double diff = fabs(a - sq); // 计算当前平方数与a的差值
        if (diff < min_diff) {
            min_diff = diff; // 更新最小差值
        }
    }
    
    // 四舍五入得到最少操作次数（最多一位小数，round无精度问题）
    int ans = round(min_diff);
    cout << ans << endl;
    
    return 0;
}

###D题：字符串回文判断
本题满分100分，平均得分四舍五入为66.66分，远低于预期。
AC code:

#include <iostream>
#include <string>
#include <cctype>
using namespace std;

int main() {
    string s;
    getline(cin, s); // 读取包含空格的整行
    
    string filtered;
    // 过滤：只保留字母和数字，并转为小写
    for (char c : s) {
        if (isalnum(c)) { // 检查是否为字母或数字
            filtered += tolower(c);
        }
    }
    
    // 判断是否为回文
    int left = 0, right = filtered.length() - 1;
    bool isPalindrome = true;
    while (left < right) {
        if (filtered[left] != filtered[right]) {
            isPalindrome = false;
            break;
        }
        left++;
        right--;
    }
    
    cout << (isPalindrome ? "YES" : "NO") << endl;
    
    return 0;
}
